create table users(
   uid int(11) primary key auto_increment,
   unique_id varchar(23) not null unique,
   name varchar(50) not null,
   email varchar(100) not null unique,
   encrypted_password varchar(80) not null,
   salt varchar(10) not null,
   created_at datetime,
   updated_at datetime null
); /** Creating Users Table **/


CREATE TABLE songs_list (email varchar(40) NOT NULL,song varchar(60) NOT NULL,song_path varchar(100) NOT NULL,genre varchar(20) DEFAULT GENERAL,artist varchar(20) DEFAULT NULL,PRIMARY KEY (email,song),
CONSTRAINT `songs_list_ibfk_1` FOREIGN KEY (email) REFERENCES users (email));
